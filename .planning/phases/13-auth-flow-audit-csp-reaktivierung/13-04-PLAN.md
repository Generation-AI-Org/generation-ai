/**
 * Content-Security-Policy builder für Website.
 * Website hat: Supabase Auth, Vercel Speed Insights, NextJS framework scripts.
 * Kein Sentry, kein Deepgram (anders als tools-app).
 * Reference: Phase 13 RESEARCH.md "CSP-Directives — Vollständige Host-Liste" website-app.
 */
export function buildCspDirectives(nonce: string, isDev: boolean): string {
  return [
    "default-src 'self'",
    // Nonce + strict-dynamic erlauben React/Next-Scripts; explizite Hosts für Speed Insights Loader.
    // 'unsafe-eval' NUR in dev (React Refresh).
    `script-src 'self' 'nonce-${nonce}' 'strict-dynamic' https://va.vercel-scripts.com${isDev ? " 'unsafe-eval'" : ""}`,
    // Tailwind v4 generiert inline styles → unsafe-inline für styles akzeptabel (für scripts NICHT).
    "style-src 'self' 'unsafe-inline'",
    "img-src 'self' blob: data:",
    "font-src 'self'",
    [
      "connect-src 'self'",
      "https://wbohulnuwqrhystaamjc.supabase.co",
      "wss://wbohulnuwqrhystaamjc.supabase.co",
      "https://va.vercel-scripts.com",
      "https://vitals.vercel-insights.com",
    ].join(" "),
    "object-src 'none'",
    "base-uri 'self'",
    "form-action 'self'",
    "frame-ancestors 'none'",
    "upgrade-insecure-requests",
  ].join("; ")
}

```typescript
import { updateSession } from "@genai/auth/middleware"
import { type NextRequest } from "next/server"
import { buildCspDirectives } from "./lib/csp"

export async function proxy(request: NextRequest) {
  // Phase-12 Pattern: Session-Refresh zuerst; response trägt Auth-Cookies.
  // Phase-13: CSP-Header wird auf DIESE Response gesetzt (Pitfall 1: keine neue Response!).
  const response = await updateSession(request)

  const nonce = Buffer.from(crypto.randomUUID()).toString("base64")
  const isDev = process.env.NODE_ENV === "development"
  const csp = buildCspDirectives(nonce, isDev)

  response.headers.set("Content-Security-Policy", csp)
  response.headers.set("x-nonce", nonce)

  return response
}

export const config = {
  matcher: [
    // api-routes + static assets ausschließen.
    // Next.js 16 CSP Pattern: prefetches nicht matchen (verhindert Cache-Probleme bei Nonce).
    {
      source:
        "/((?!api|_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)",
      missing: [
        { type: "header", key: "next-router-prefetch" },
        { type: "header", key: "purpose", value: "prefetch" },
      ],
    },
  ],
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Website HTML | CSP kontrolliert welche Scripts/Styles/Fonts/Connections der Browser lädt — primäre XSS-Defense. |
| Proxy middleware → Next.js response | Nonce wird pro-Request generiert; Next.js muss dieselbe Nonce aus `x-nonce`-Header lesen und auf Framework-Scripts anwenden. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-11 | Tampering (XSS) | Inline-script injection via user input | mitigate | `script-src 'nonce-{n}' 'strict-dynamic'` blockt inline scripts ohne matching nonce. `'unsafe-inline'` ENTFERNT aus script-src. |
| T-13-12 | Tampering | Clickjacking via iframe embed | mitigate | `frame-ancestors 'none'` + X-Frame-Options: DENY (belt + suspenders) |
| T-13-13 | Information Disclosure | Form submit zu fremder Domain (credential phishing) | mitigate | `form-action 'self'` (Supabase-Auth-Forms gehen an `/api/auth/*` same-origin) |
| T-13-14 | Availability | CSP blockt eigene Scripts → Site kaputt | mitigate | `'strict-dynamic'` + Speed-Insights-Host explicit + Preview-Check (Task 3 Decision-Gate) vor Prod-Deploy |
| T-13-15 | Tampering | Base-URL-Hijack | mitigate | `base-uri 'self'` |
| T-13-16 | Tampering (Nonce reuse) | Gleiche Nonce auf mehreren Requests → bypass | mitigate | `crypto.randomUUID()` pro Request; response-caching deaktiviert durch prefetch-skip im matcher |
| T-13-17 | Availability | Prefetch cached alte Nonce → 2. Request schlägt fehl | mitigate | matcher.missing inkludiert `next-router-prefetch` + `purpose:prefetch` |
</threat_model>
