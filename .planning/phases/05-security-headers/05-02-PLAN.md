import { createClient } from '@/lib/supabase/proxy'
import { NextResponse, type NextRequest } from 'next/server'

export async function proxy(request: NextRequest) {
  let response = NextResponse.next({
    request: { headers: request.headers },
  })

  const { supabase, response: updatedResponse } = createClient(request, response)
  response = updatedResponse

  await supabase.auth.getUser()

  return response
}

export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)',
  ],
}

  <name>Task 2: Nonce-basierte CSP in proxy.ts implementieren</name>
  <files>apps/tools-app/app/proxy.ts</files>
  <action>
Erweitere proxy.ts um Nonce-basierte CSP-Generation:

1. Generiere Nonce pro Request:
   ```typescript
   const nonce = Buffer.from(crypto.randomUUID()).toString('base64')
   ```

2. Baue CSP-Header (Report-Only erst zum Testen):
   ```typescript
   const isDev = process.env.NODE_ENV === 'development'
   const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL || 'https://wbohulnuwqrhystaamjc.supabase.co'
   
   const cspDirectives = [
     "default-src 'self'",
     `script-src 'self' 'nonce-${nonce}' 'strict-dynamic'${isDev ? " 'unsafe-eval'" : ''}`,
     `style-src 'self' 'nonce-${nonce}'${isDev ? " 'unsafe-inline'" : ''}`,
     "img-src 'self' blob: data: https://logo.clearbit.com",
     "font-src 'self'",
     `connect-src 'self' ${supabaseUrl}`,
     "object-src 'none'",
     "base-uri 'self'",
     "form-action 'self'",
     "frame-ancestors 'none'",
     "upgrade-insecure-requests",
   ]
   const cspHeader = cspDirectives.join('; ')
   ```

3. Setze Nonce als Request-Header (fuer Server Components):
   ```typescript
   requestHeaders.set('x-nonce', nonce)
   ```

4. Setze CSP-Report-Only Header auf Response:
   ```typescript
   response.headers.set('Content-Security-Policy-Report-Only', cspHeader)
   ```

5. KRITISCH: Behalte die bestehende Supabase Auth-Logik (createClient, getUser) bei!

Die Reihenfolge:
1. Nonce generieren
2. Request Headers vorbereiten (inkl. x-nonce)
3. Response erstellen
4. Supabase Auth ausfuehren (kann Response modifizieren)
5. CSP Header auf finale Response setzen
6. Response zurueckgeben
