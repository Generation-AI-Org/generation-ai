# Plan 22-04 — PartnerTabSystem + PartnerTabContent

## Objective

Create the 4-Tab system for the partner page:
- `PartnerTabSystem` — ARIA-compliant tab rail with URL-param sync via `pushState`
- `PartnerTabContent` — 4 tab panel variants (Unternehmen/Stiftungen/Hochschulen/Initiativen), each with value-prop + Formate list + CTA

## Context

- D-01: 4-Tab-System instead of anchor sections — canonical
- D-02: URL parameter `?typ=...` not hash fragments — pushState, no full-reload
- D-03: No scroll-reset on tab switch — smooth fade-in transition
- D-04: Default tab "Unternehmen", no URL push on default landing (no history spam)
- UI-SPEC Tab System — Interaction Contract (lines 132-172)

### Tab Types and Values

```
Type slug → Display label → H2 → Formate
unternehmen → "Unternehmen" → "Employer Branding trifft echte Zielgruppe." → Masterclasses · Speaker Sessions · Sponsoring
stiftungen  → "Stiftungen"  → "Impact mit Substanz." → Projekt/Programm-Förderung · Stipendien · institutionelle Förderung
hochschulen → "Hochschulen" → "Praxis direkt an die Uni." → Gastvorträge · Career-Service-Integration · Lehrstuhl-Kooperationen
initiativen → "Initiativen" → "Gemeinsam mehr erreichen." → Co-hosted Events · Cross-Promotion · geteilte Speaker
```

### ARIA Pattern (from UI-SPEC lines 137-148)
- Container: `role="tablist"` with `aria-label="Partnertypen"`
- Tab: `role="tab"` + `aria-selected={true|false}` + `aria-controls="{slug}-panel"` + `id="{slug}-tab"`
- Panel: `role="tabpanel"` + `id="{slug}-panel"` + `aria-labelledby="{slug}-tab"` + `tabIndex={0}`
- Hidden panels: `hidden` HTML attribute (not CSS class)

### Mobile: horizontal scroll tab rail
- `overflow-x-auto scrollbar-hide flex flex-nowrap gap-1 px-6`
- Each tab: `min-h-[44px]` touch target
- Active tab scrolls into view on mount if deep-linked

### Tab Visual States (UI-SPEC lines 154-160)
- Inactive: `text-text-muted` + transparent bg + `font-mono font-bold text-[14px] tracking-[0.02em]`
- Active: `text-text` + `border-b-2 border-[var(--accent)]`
- Hover: `text-text-secondary`, `transition duration-[var(--dur-fast)]`
- Focus-visible: `outline-2px solid var(--text)` + `outline-offset-2px` + `border-radius-4px`

### Keyboard Navigation
- Arrow Left/Right navigates between tabs
- Enter/Space activates focused tab

### Tab Content Transition (UI-SPEC lines 161-165)
- Fade-out active panel (opacity 0, 150ms), fade-in new (opacity 1, y: 8→0, 300ms)
- `useReducedMotion`: skip animation, swap instantly
- Framer Motion `AnimatePresence` mode="wait" OR CSS transition + key prop

```tsx
'use client'

import { useCallback, useEffect, useRef } from "react"
import { useSearchParams } from "next/navigation"
import { PartnerTabContent } from "./partner-tab-content"
import type { PartnerTyp } from "./partner-tab-content"

const TABS: Array<{ slug: PartnerTyp; label: string }> = [
  { slug: 'unternehmen', label: 'Unternehmen' },
  { slug: 'stiftungen', label: 'Stiftungen' },
  { slug: 'hochschulen', label: 'Hochschulen' },
  { slug: 'initiativen', label: 'Initiativen' },
]

const VALID_SLUGS = new Set<string>(TABS.map((t) => t.slug))

interface PartnerTabSystemProps {
  initialTyp?: string
}

export function PartnerTabSystem({ initialTyp }: PartnerTabSystemProps) {
  // Resolve initial active tab from SSR prop (D-04: no pushState on default)
  const resolvedInitial: PartnerTyp =
    initialTyp && VALID_SLUGS.has(initialTyp)
      ? (initialTyp as PartnerTyp)
      : 'unternehmen'

  // We use a ref-driven approach for active tab to avoid re-render on pushState
  // and to support SSR hydration parity. Parent passes initialTyp from searchParams.
  const [activeTyp, setActiveTyp] = usePartnerTabState(resolvedInitial)

  const tabRailRef = useRef<HTMLDivElement>(null)

  // Scroll active tab into view (for deep-linked ?typ= mounts)
  useEffect(() => {
    const rail = tabRailRef.current
    if (!rail) return
    const activeTab = rail.querySelector('[aria-selected="true"]') as HTMLElement | null
    if (activeTab) {
      activeTab.scrollIntoView({ inline: 'center', behavior: 'smooth' })
    }
  }, [activeTyp])

  // Keyboard navigation: Arrow Left/Right
  const handleKeyDown = useCallback(
    (e: React.KeyboardEvent<HTMLDivElement>, currentIndex: number) => {
      let nextIndex: number | null = null
      if (e.key === 'ArrowRight') {
        nextIndex = (currentIndex + 1) % TABS.length
      } else if (e.key === 'ArrowLeft') {
        nextIndex = (currentIndex - 1 + TABS.length) % TABS.length
      } else if (e.key === 'Home') {
        nextIndex = 0
      } else if (e.key === 'End') {
        nextIndex = TABS.length - 1
      }

      if (nextIndex !== null) {
        e.preventDefault()
        const nextSlug = TABS[nextIndex]!.slug
        setActiveTyp(nextSlug)
        // Focus the next tab button
        const nextBtn = tabRailRef.current?.querySelector(
          `[id="${nextSlug}-tab"]`,
        ) as HTMLButtonElement | null
        nextBtn?.focus()
      }
    },
    [setActiveTyp],
  )

  return (
    <section
      aria-label="Partnertypen auswählen"
      data-section="partner-tab-system"
      className="relative bg-bg"
    >
      {/* Tab rail */}
      <div
        ref={tabRailRef}
        role="tablist"
        aria-label="Partnertypen"
        className="overflow-x-auto scrollbar-hide border-b border-border"
      >
        <div className="flex flex-nowrap gap-1 px-6 mx-auto max-w-4xl">
          {TABS.map((tab, index) => {
            const isActive = tab.slug === activeTyp
            return (
              <button
                key={tab.slug}
                id={`${tab.slug}-tab`}
                role="tab"
                aria-selected={isActive}
                aria-controls={`${tab.slug}-panel`}
                onClick={() => setActiveTyp(tab.slug)}
                onKeyDown={(e) => handleKeyDown(e, index)}
                tabIndex={isActive ? 0 : -1}
                className={[
                  'min-h-[44px] px-4 py-3 font-mono font-bold text-[14px] tracking-[0.02em] whitespace-nowrap transition-colors border-b-2',
                  'focus-visible:outline-2 focus-visible:outline-solid focus-visible:outline-text focus-visible:outline-offset-2 focus-visible:rounded-[4px]',
                  isActive
                    ? 'text-text border-[var(--accent)]'
                    : 'text-text-muted border-transparent hover:text-text-secondary',
                ].join(' ')}
                style={
                  isActive
                    ? { borderBottomColor: 'var(--accent)', color: 'var(--accent)' }
                    : {}
                }
              >
                {tab.label}
              </button>
            )
          })}
        </div>
      </div>

      {/* Tab panels */}
      <PartnerTabContent activeTyp={activeTyp} />
    </section>
  )
}

/**
 * Custom hook: manages active tab state + URL pushState sync.
 * - On mount: uses initialTyp (SSR-resolved, no pushState for default D-04)
 * - On tab click: updates state + pushState with ?typ={slug}
 */
function usePartnerTabState(initial: PartnerTyp) {
  const [activeTyp, setActiveTypRaw] = useStateWithRef(initial)

  const setActiveTyp = useCallback(
    (slug: PartnerTyp) => {
      setActiveTypRaw(slug)
      // D-02: push URL param without full navigation (no scroll reset D-03)
      window.history.pushState(null, '', `?typ=${slug}`)
    },
    [setActiveTypRaw],
  )

  return [activeTyp, setActiveTyp] as const
}

function useStateWithRef<T>(initial: T) {
  const [state, setState] = require('react').useState<T>(initial)
  return [state, setState] as const
}
```
