```typescript
const audioCtx = new AudioContext()
audioCtxRef.current = audioCtx

if (audioCtx.state === 'suspended') {
  await audioCtx.resume()
}

const analyser = audioCtx.createAnalyser()
analyser.fftSize = 32  // Gibt frequencyBinCount = 16
analyser.smoothingTimeConstant = 0.8
analyserRef.current = analyser

const source = audioCtx.createMediaStreamSource(stream)
source.connect(analyser)
// NICHT an destination connecten (kein Mic-Echo!)

const bufferLength = analyser.frequencyBinCount
dataArrayRef.current = new Uint8Array(bufferLength)
```

**Visualisierung Loop starten:**
```typescript
function startVisualizationLoop() {
  const loop = () => {
    if (!analyserRef.current || !dataArrayRef.current) return
    
    analyserRef.current.getByteFrequencyData(dataArrayRef.current)
    
    // Ersten 12 Bins nehmen, auf 0-1 normalisieren
    const levels = Array.from({ length: 12 }, (_, i) => {
      return (dataArrayRef.current![i] ?? 0) / 255
    })
    
    setAudioLevels(levels)
    animFrameRef.current = requestAnimationFrame(loop)
  }
  
  animFrameRef.current = requestAnimationFrame(loop)
}
```

**In startRecording() aufrufen:**
```typescript
// Nach Analyser Setup
startVisualizationLoop()
```

**In stopRecording() Cleanup:**
```typescript
// Animation stoppen
if (animFrameRef.current) {
  cancelAnimationFrame(animFrameRef.current)
  animFrameRef.current = null
}

// Audio Context schliessen
if (audioCtxRef.current) {
  audioCtxRef.current.close()
  audioCtxRef.current = null
}
analyserRef.current = null

// Levels zuruecksetzen
setAudioLevels(new Array(12).fill(0))
```

**Cleanup im useEffect erweitern:**
```typescript
useEffect(() => {
  return () => {
```
