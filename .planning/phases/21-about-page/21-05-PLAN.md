import { motion, useReducedMotion } from "motion/react"

export function AboutMitmachCTASection() {

  return (
    <section
      id="mitmach"

        {/* H2 */}
        <motion.h2
          {...fadeIn}
          id="about-mitmach-heading"

          style={{ fontSize: "var(--fs-h2)", lineHeight: "var(--lh-headline)" }}

          Bock, mitzumachen?
        </motion.h2>

        {/* Body */}

          {...fadeIn}
