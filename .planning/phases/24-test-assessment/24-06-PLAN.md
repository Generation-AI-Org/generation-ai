    Create `apps/website/lib/shiki.ts`:
    ```ts
    // Server-side shiki singleton. Call `highlightCode(code, lang)` to get HTML string.
    // Used by page loaders (server components) to pre-render code blocks for question widgets.
    // Theme: github-dark (matches site default). Light-theme swap is out of phase-24 scope.
    import { createHighlighter, type Highlighter } from 'shiki'

    let highlighterPromise: Promise<Highlighter> | null = null

    async function getHighlighter(): Promise<Highlighter> {
      if (!highlighterPromise) {
        highlighterPromise = createHighlighter({
          themes: ['github-dark'],
          langs: ['javascript', 'typescript', 'python', 'json', 'bash', 'plaintext'],
        })
      }
      return highlighterPromise
    }

    export async function highlightCode(code: string, lang: string): Promise<string> {
      const hl = await getHighlighter()
      // Fallback to plaintext if lang is unknown
      const safeLang = hl.getLoadedLanguages().includes(lang as never) ? lang : 'plaintext'
      return hl.codeToHtml(code, { lang: safeLang, theme: 'github-dark' })
    }
    ```

    This file runs server-side only (shiki adds zero bytes to client bundle when all callers are server components).
  </action>
