  <read_first>

  <action>
    Create `packages/e2e-tools/tests/test-assessment.spec.ts`:
    ```ts
    import { test, expect } from '@playwright/test'

    const BASE_URL = process.env.E2E_BASE_URL ?? 'http://localhost:3000'
    const TEST_URL = `${BASE_URL}/test`

    test.describe('/test AI Literacy Assessment', () => {
      test.beforeEach(async ({ page }) => {
        // Skip Terminal-Splash overlay (same pattern as join.spec.ts)
        await page.addInitScript(() => {
          try {
            sessionStorage.setItem('terminal-splash-seen', 'true')
          } catch {
            // ignore
          }
        })
      })

      test('hero page loads with title and Start button', async ({ page }) => {
        const response = await page.goto(TEST_URL)
        expect(response?.status()).toBe(200)
        await expect(page).toHaveTitle(/AI Literacy Test/)
        await expect(page.getByRole('heading', { level: 1 })).toContainText('Wo stehst du wirklich')
        await expect(page.getByRole('link', { name: 'Test starten' })).toBeVisible()
      })

      test('Start button navigates to Aufgabe 1', async ({ page }) => {
        await page.goto(TEST_URL)
        await page.getByRole('link', { name: 'Test starten' }).click()
        await page.waitForURL(/\/test\/aufgabe\/1$/)
        await expect(page.locator('[data-widget-type]')).toBeVisible()
      })

      test('at least 3 distinct widget types appear across the 10 questions', async ({ page }) => {
        await page.goto(`${BASE_URL}/test/aufgabe/1`)
        const seenTypes = new Set<string>()

        for (let i = 1; i <= 10; i++) {
          await page.waitForURL(new RegExp(`/test/aufgabe/${i}$|/test/ergebnis$`))
          if (page.url().includes('/test/ergebnis')) break

          const widget = page.locator('[data-widget-type]').first()
          await widget.waitFor({ state: 'visible', timeout: 5000 })
          const widgetType = await widget.getAttribute('data-widget-type')
          if (widgetType) seenTypes.add(widgetType)

          // Minimal interaction per widget-type to enable Next:
          const nextBtn = page.getByRole('button', { name: 'Nächste Aufgabe' })
          if (widgetType === 'pick' || widgetType === 'mc' || widgetType === 'best-prompt') {
            await page.locator('[role="radio"]').first().click()
          } else if (widgetType === 'side-by-side') {
            await page.locator('[role="radiogroup"] [role="radio"]').first().click()
          } else if (widgetType === 'spot') {
            await page.locator('[role="listbox"] [role="option"]').first().click()
          } else if (widgetType === 'match') {
            // Touch-mode (select) OR drag. Use all selects if present.
            const selects = page.locator('[data-widget-type="match"] select')
            const count = await selects.count()
            for (let s = 0; s < count; s++) {
              await selects.nth(s).selectOption({ index: 1 })
            }
          } else if (widgetType === 'confidence') {
            // Slider has a default value — Next should already be enabled.
          } else if (widgetType === 'fill') {
            const selects = page.locator('[data-widget-type="fill"] select')
            const count = await selects.count()
            for (let s = 0; s < count; s++) {
              await selects.nth(s).selectOption({ index: 1 })
            }
          } else if (widgetType === 'rank') {
            // Just click Next — initial order counts as an answer (rank is always ready once mounted).
            // If the widget gates on user interaction, we'd need keyboard nav here — accept FAIL and investigate.
          }

          await nextBtn.click()
        }

        expect(seenTypes.size).toBeGreaterThanOrEqual(3)
      })

      test('completing all 10 questions lands on /test/ergebnis with level + primary CTA', async ({ page }) => {
        await page.goto(`${BASE_URL}/test/aufgabe/1`)

        // Quickly blast through all questions picking first options
        for (let i = 1; i <= 10; i++) {
          const widget = page.locator('[data-widget-type]').first()
          await widget.waitFor({ state: 'visible', timeout: 5000 })
          const widgetType = await widget.getAttribute('data-widget-type')

          if (widgetType === 'pick' || widgetType === 'mc' || widgetType === 'best-prompt' || widgetType === 'side-by-side') {
            await page.locator('[role="radio"]').first().click()
          } else if (widgetType === 'spot') {
            await page.locator('[role="option"]').first().click()
          } else if (widgetType === 'match' || widgetType === 'fill') {
            const selects = page.locator(`[data-widget-type="${widgetType}"] select`)
            const count = await selects.count()
            for (let s = 0; s < count; s++) await selects.nth(s).selectOption({ index: 1 })
          }

          await page.getByRole('button', { name: 'Nächste Aufgabe' }).click()
          // Checkpoint after 5 — wait it out (1.5s + buffer)
          if (i === 5) await page.waitForTimeout(1800)
        }

        await page.waitForURL(/\/test\/ergebnis$/)
        await expect(page.getByRole('heading', { level: 1 })).toContainText(/Level \d/)
        const primaryCta = page.getByRole('link', { name: /Jetzt beitreten/ })
        await expect(primaryCta).toBeVisible()
        const href = await primaryCta.getAttribute('href')
        expect(href).toMatch(/^\/join\?pre=(neugieriger|einsteiger|fortgeschritten|pro|expert)&source=test&skills=/)
      })

      test('Skill Radar figure exists on results page', async ({ page }) => {
        // Re-run: start at aufgabe 1 since assessment state is per-session
        await page.goto(`${BASE_URL}/test/aufgabe/1`)
        // Fast-forward through all 10
        for (let i = 1; i <= 10; i++) {
          const widget = page.locator('[data-widget-type]').first()
          await widget.waitFor({ state: 'visible', timeout: 5000 })
          const widgetType = await widget.getAttribute('data-widget-type')
          if (widgetType === 'pick' || widgetType === 'mc' || widgetType === 'best-prompt' || widgetType === 'side-by-side') {
            await page.locator('[role="radio"]').first().click()
          } else if (widgetType === 'spot') {
            await page.locator('[role="option"]').first().click()
          } else if (widgetType === 'match' || widgetType === 'fill') {
            const selects = page.locator(`[data-widget-type="${widgetType}"] select`)
            const count = await selects.count()
            for (let s = 0; s < count; s++) await selects.nth(s).selectOption({ index: 1 })
          }
          await page.getByRole('button', { name: 'Nächste Aufgabe' }).click()
          if (i === 5) await page.waitForTimeout(1800)
        }
        await page.waitForURL(/\/test\/ergebnis$/)
        await expect(page.locator('[data-testid="skill-radar"]')).toBeVisible()
      })

      test('/test/aufgabe/1 loads standalone and renders Aufgabe layout', async ({ page }) => {
        // Direct deep-link: should redirect to /test per URL-state guard (D-11) ONLY if the user had
        // not answered previous questions — but aufgabe/1 has no prior questions, so it loads.
        const response = await page.goto(`${BASE_URL}/test/aufgabe/1`)
        expect(response?.status()).toBe(200)
        await expect(page.locator('[data-widget-type]')).toBeVisible()
        await expect(page.getByText(/Aufgabe 1\/10/)).toBeVisible()
      })

      test('/test/ergebnis without completed test shows NoResultFallback', async ({ page }) => {
        await page.goto(`${BASE_URL}/test/ergebnis`)
        await expect(page.getByText(/Kein Ergebnis vorhanden/)).toBeVisible()
        await expect(page.getByRole('link', { name: 'Test starten' })).toBeVisible()
      })
    })
    ```

    Notes:
    - Tests 3 and 4 walk the full flow; test 5 re-walks because the assessment state resets on route change (provider is at /test layout, which re-mounts on navigation). This is acceptable Playwright cost.
    - Test "Skill Radar figure exists" is somewhat redundant with test 4 — keep if it passes cheap, else merge.
    - If test 3 or 4 turn flaky in Playwright due to checkpoint timing, increase `waitForTimeout(1800)` to 2200 ms.
