   import { createContext, useContext, useState, ReactNode } from 'react'
   import type { ChatContext } from '@/lib/types'

   interface ChatContextValue {
     chatContext: ChatContext | null
     setChatContext: (ctx: ChatContext | null) => void

     type AnalyticsEvent = {

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → /api/chat | Neues `context`-Feld wird von Client serialisiert — untrusted input |
| Analytics-Payload | `route` via `window.location.pathname` — potentiell user-controlled (Custom-URL-Manipulation) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-15-04 | T (Tampering) | /api/chat context field | mitigate | Strict type-check auf slug/title/type/summary als strings, summary.length < 2000; invalid → ignorieren (kein Error). `sanitizeUserInput()` auf alle Felder vor LLM-Prefix-Konstruktion. |
| T-15-05 | I (Information Disclosure) | Context in LLM-Prompt | accept | Context enthält nur öffentliche Tool-Metadata (gleiche Daten wie auf /[slug] gerendert). Kein PII, kein Secret. |
| T-15-06 | R (Repudiation) | Analytics-Event | accept | Fire-and-forget Tracking, keine auth-gated Actions, kein Compliance-Impact. |
| T-15-07 | D (DoS) | Context Length | mitigate | summary-Cap auf 2000 chars → Token-Budget geschützt, keine 100kB-Payloads an Gemini. |
| T-15-08 | E (Elevation) | Prompt Injection via context | mitigate | `sanitizeUserInput()` filtert Kontroll-Tokens. Context stammt aus Supabase-DB (admin-curated), nicht user-generated — Injection-Risiko niedrig. Bei zukünftigem user-editable content: re-evaluieren. |
</threat_model>
